26        function obj=Cylinder(varargin)
27            obj.R=0.02;
28            obj.M=0.003;

46        function ForceReaction=ForceReactionCalc(obj,CollisionCondition,IndexFext,ThumbFext)

48                if(CollisionCondition(1)==1)
