1classdef Cylinder < handle & matlab.System 
  2    properties 
  3       R;%Radius of the cylindrical object in meters
  4       M;%Mass of the object in Kg
  5       g;%Acceleration of gravity in m/s^2
  6       Us;%Static coefficient of friction
  7       Uk;%Kinetic coefficient of friction
  8       Iz;%Inertia of the cylindrical object in Z
  9       F_s_x;%Static friction force in X
 10       F_s_y;%Static friction force in Y
 11       F_k_x;%Kinetic friction force in X
 12       F_k_y;%Kinetic friction force in Y
 13       T_s;%Static friction torque see derivation notes to verify the expression
 14       T_k;%Kinetic friction torque see derivation notes to verify the expression
 15       Position_o_G3;%Position from frame 0 to the centre of gravity of the cylinder
 16       Position_G3_P3_frame_cylinder=[0;0];%Position to second point of the cylinder
 17    end
 18    methods  
 19    %............................................................................
 20    %............................................................................
 21    %Constructor
 22        function obj=Cylinder(varargin)
 23            obj.R=0.02;
 24            obj.M=0.003;
 25            obj.g=9.81;
 26            obj.Us=0.5;
 27            obj.Uk=0.48;
 28            obj.Iz=(obj.M*obj.R^2)/2;
 29            obj.F_s_x=obj.M*obj.g*obj.Us;
 30            obj.F_s_y=obj.M*obj.g*obj.Us;
 31            obj.F_k_x=obj.M*obj.g*obj.Uk;
 32            obj.F_k_y=obj.M*obj.g*obj.Uk;
 33            obj.T_s=(2/3)*obj.Us*obj.R*obj.M*obj.g;
 34            obj.T_k=(2/3)*obj.Uk*obj.R*obj.M*obj.g;
 35            obj.Position_o_G3=[0;0];
 36            obj.Position_G3_P3_frame_cylinder=[0;obj.R];
 37        end
 38    %............................................................................
 39    %............................................................................
 40    %Evaluates if the external force is greater than the static
 41    %friction forces or torques
 42        function FrictionCondition=NonLinearFriction(obj,GenCordExt,Election)
 43            %Evaluates wich static friction have to be compared
 44            if (strcmp(Election,'F_s_x')==1)
 45                ForceEvaluation=obj.F_s_x;
 46            elseif(strcmp(Election,'F_s_y')==1)
 47                ForceEvaluation=obj.F_s_y;
 48            else
 49                ForceEvaluation=obj.T_s;
 50            end
 51            %Condition checker
 52            if(GenCordExt<=ForceEvaluation)
 53                FrictionCondition=0;%if the Friction condition is 0 there is no movement
 54            else
 55                FrictionCondition=1;%if the Friction condition is 1 there is movement
 56            end
 57        end
 58    %............................................................................
 59    %............................................................................
 60    %Calculates the acceleration components in X and Y for the centre of
 61    %gravity of the object and the angular acceleration
 62        function GeneralCoordinatesDoubleDot = Equation_Of_Motion(obj,GenCordExt)
 63            GeneralCoordinatesDoubleDot=zeros(3);
 64            %Ecuation of motion for X coordinates
 65            SignOposite=-1*sign(GenCordExt);
 66            if(NonLinearFriction(obj,abs(GenCordExt(1)),'F_s_x')== 1)
 67                GeneralCoordinatesDoubleDot(1)= (GenCordExt(1)+ SignOposite(1)*obj.F_k_x)/obj.M;
 68            else
 69                GeneralCoordinatesDoubleDot(1)= 0;
 70            end
 71        %Ecuation of motion for y coordinates
 72            if(NonLinearFriction(obj,abs(GenCordExt(2)),'F_s_y')== 1)
 73                GeneralCoordinatesDoubleDot(2)= (GenCordExt(2)+ SignOposite(2)*obj.F_k_y)/obj.M;
 74            else
 75                GeneralCoordinatesDoubleDot(2)= 0;
 76            end
 77        %Ecuation of motion for Tetha coordinates
 78            if(NonLinearFriction(obj,abs(GenCordExt(3)),'T_s')== 1)
 79                GeneralCoordinatesDoubleDot(3)= (GenCordExt(3)+ SignOposite(3)*obj.T_k)/obj.Iz;
 80            else
 81                GeneralCoordinatesDoubleDot(3)= 0;
 82            end
 83        end
 84    %............................................................................
 85    %............................................................................
 86    %Absolute positions for G3 and P3 updated from frame 0
 87        function Positions = PosCalculation(obj,angle,xPositionG3,yPositionG3)
 88            Rotation_o_3=[cos(angle),-sin(angle);sin(angle),cos(angle)];
 89            Position_G3_P3_frame_0=Rotation_o_3*obj.Position_G3_P3_frame_cylinder;
 90            Position_o_P3=obj.Position_o_G3 + Position_G3_P3_frame_0;
 91            Positions=[obj.Position_o_G3,Position_o_P3];
 92        end
 93    %............................................................................
 94    %............................................................................
 95    %Absolute velocities for G3 and P3 updated from frame 0
 96        function Velocities = VelCalculation (obj,angle,angleDot,xDotG3,yDotG3)
 97            Velocity_o_G3_frame_0=[xDotG3;yDotG3];
 98            Velocity_o_P3_frame_0=[xDotG3-obj.R*cos(angle)*angleDot;yDotG3-obj.R*sin(angle)*angleDot];
 99            Velocities=[Velocity_o_G3_frame_0,Velocity_o_P3_frame_0];
100        end
101    %............................................................................
102    %............................................................................
103    %Absolute Accelerations for G3 and P3 updated from frame 0
104        function Accelerations = AccCalculation (obj,angle,angleDot,angleDoubleDot,xDoubleDot,yDoubleDot)
105            Acceleration_o_G3_frame_0=[xDoubleDot;yDoubleDot];
106            Acceleration_o_P3_frame_0=[xDoubleDot+obj.R*sin(angle)*(angleDot^2)-obj.R*cos(angle)*angleDoubleDot;...
107                                       yDoubleDot-obj.R*cos(angle)*(angleDot^2)-obj.R*sin(angle)*angleDoubleDot];
108            Accelerations=[Acceleration_o_G3_frame_0,Acceleration_o_P3_frame_0];
109        end
110    end
111    methods (Access = protected)
112    %............................................................................
113    %............................................................................
114    %Function that is called in Simulink to calculate updated
115    %positions,velocities and accelerations for the cylinder

116        function CylinderResults=stepImpl(obj,FextX,FextY,TorqueExt,angleDot,angle,...
117                                                        xDotG3,yDotG3,xPositionG3,yPositionG3)
118            GenCordExt=[FextX,FextY,TorqueExt];
119            GeneralCoordinatesDoubleDot = Equation_Of_Motion(obj,GenCordExt);
120            xDoubleDot = GeneralCoordinatesDoubleDot(1);
121            yDoubleDot = GeneralCoordinatesDoubleDot(2);
122            angleDoubleDot = GeneralCoordinatesDoubleDot(3);
123            Positions = PosCalculation(obj,angle,xPositionG3,yPositionG3);
124            Velocities = VelCalculation (obj,angle,angleDot,xDotG3,yDotG3);
125            Accelerations = AccCalculation (obj,angle,angleDot,angleDoubleDot,xDoubleDot,yDoubleDot);
126            CylinderResults = [Accelerations(1),Accelerations(2),angleDoubleDot,obj.R,Positions(:,1)];
127        end

128    end
129end
