1classdef ForceNormal < handle
 2    % ForceNormal class
 3    % Represents a Force vector [fn,ft] acting on a Finger at a distance
 4    % d from a joint. fn represents the Force normal to the finger and
 5    % ft the tangential Force due to friction miu.
 6    properties (SetAccess = private)
 7        fv % Force vector [fn;ft]
 8        d % Distance from joint to Force (m)
 9    end
10    methods

11        function obj = ForceNormal(fn,d,miu)
12            % Create a new ForceFinger object
13            obj.fv = [fn;miu*fn];
14            obj.d = d;
15        end

16        function setD(obj,d)
17            % Set the distance from the joint to the Force
18            obj.d = d;
19        end
20        function setForceV(obj,fn,miu)
21            % Set the Force vector [fx;fy]
22            obj.fv = [fn;miu*fn];
23        end
24    end
25end
