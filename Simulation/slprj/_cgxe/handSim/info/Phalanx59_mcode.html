 2% Represents the phalanx of a robotic hand finger.
 3% Contains the properties that represent the physical properties of a
 4% robotic hand phalanx, as well as the external forces acting on it.
 5% Change log:

10classdef Phalanx < handle
11% Phalanx Class
12% Represents the phalanx of a robotic hand finger.
13% Contains the properties that represent the physical properties of a
14% robotic hand phalanx, as well as the external forces acting on it.
15    properties (SetAccess = immutable)
16        % Physical Phalanx properties
17        k = 1;      % Joint stiffness (Nm/rad)
18        d = 0.01    % Joint damping coefficient (Nm/(rad/s))
19        l = 0.1;    % Link length (m)
20        r = 0.02;   % Pulley radius (m)
21        m = 0.02;   % Link mass (kg)
22        I           % Link z-inertia
23    end
24    properties
25        % Physical phalanx properties
26        miuC = 0.8; % Coefficient of friction between phalanx and contact object.
27        miuE = 0.8; % Coefficient of friction between phalanx and disturbance object.% Kinematic properties
28        theta0 = pi/10; % Joint rest angle (rad)
29        theta = pi/10; % Joint angle (rad)
30        thetaDot = 0; % First derivative (velocity) of joint angle (rad/s)
31        thetaDotDot = 0; % Second derivative (acceleration) of joint angle (rad/s^2)
32        % Contact Kinematics properties
33        a = 0.1; % Distance from joint to contact force (m)
34        fc % Contact Force
35        % Disturbance Kinematics properties
36        b = 0.05; % Distance from joint to disturbance force (m)
37        fe % Disturbance Force
38    end
39    methods

40        function obj = Phalanx(k, d, l, r, m, theta, linkNum)
41            % Constructor with parameter initialization
42            % @param linkNum the link number in the chain.
43            
44            % Custom initialization
45            if(nargin ~= 0)
46                obj.k = k;
47                obj.d = d;
48                obj.l = l;
49                obj.r = r;
50                obj.m = m;
51                obj.theta = theta;
52                % Compute intertia matrix depending on which part of the
53                % chain the link is.
54                if (linkNum>1)
55                    obj.I = obj.m*(obj.l^2)/12;
56                else
57                    obj.I = obj.m*(obj.l^2)/3;
58                end
59            else
60                obj.I = obj.m*(obj.l^2)/3;
61            end
62            obj.theta0 = obj.theta;
63            obj.thetaDot = 0;
64            obj.thetaDotDot = 0;
65            obj.fc = ForceNormal(0,obj.a,obj.miuC);
66            obj.fe = ForceNormal(0,obj.b,obj.miuE);
67        end

68    end
69end
