1classdef Cylinder < handle & matlab.System 
  2    properties 
  3       R;%Radius of the cylindrical object in meters
  4       M;%Mass of the object in Kg
  5       g;%Acceleration of gravity in m/s^2
  6       Us;%Static coefficient of friction
  7       Uk;%Kinetic coefficient of friction
  8       Urs;%Static rotational coefficient of friction
  9       Urk;%Kinetic rotational coefficient of friction
 10       Iz;%Inertia of the cylindrical object in Z
 11       F_s_x;%Static friction force in X
 12       F_s_y;%Static friction force in Y
 13       F_k_x;%Kinetic friction force in X
 14       F_k_y;%Kinetic friction force in Y
 15       T_s;%Static friction torque see derivation notes to verify the expression
 16       T_k;%Kinetic friction torque see derivation notes to verify the expression
 17       Position_o_G3;%Position from frame 0 to the centre of gravity of the cylinder
 18       Position_G3_P3_frame_cylinder=[0;0];%Position to second point of the cylinder
 19    end
 20    methods  
 21    %............................................................................
 22    %............................................................................
 23    %Constructor
 24        function obj=Cylinder(varargin)
 25            obj.R=0.02;
 26            obj.M=0.003;
 27            obj.g=9.81;
 28            obj.Us=0.5;
 29            obj.Uk=0.48;
 30            obj.Urs=0.5;
 31            obj.Urk=0.48;
 32            obj.Iz=(obj.M*obj.R^2)/2;
 33            obj.F_s_x=obj.M*obj.g*obj.Us;
 34            obj.F_s_y=obj.M*obj.g*obj.Us;
 35            obj.F_k_x=obj.M*obj.g*obj.Uk;
 36            obj.F_k_y=obj.M*obj.g*obj.Uk;
 37            obj.T_s=(2/3)*obj.Urs*obj.R*obj.M*obj.g;
 38            obj.T_k=(2/3)*obj.Urk*obj.R*obj.M*obj.g;
 39            obj.Position_o_G3=[0;0];
 40            obj.Position_G3_P3_frame_cylinder=[0;obj.R];
 41        end
 42    %............................................................................
 43    %............................................................................
 44    %Evaluates if the external force is greater than the static
 45    %friction forces or torques
 46        function FrictionCondition=NonLinearFriction(obj,GenCordExt,Election)
 47            %Evaluates wich static friction have to be compared
 48            if (strcmp(Election,'F_s_x')==1)
 49                ForceEvaluation=obj.F_s_x;
 50            elseif(strcmp(Election,'F_s_y')==1)
 51                ForceEvaluation=obj.F_s_y;
 52            else
 53                ForceEvaluation=obj.T_s;
 54            end
 55            %Condition checker
 56            if(GenCordExt<=ForceEvaluation)
 57                FrictionCondition=0;%if the Friction condition is 0 there is no movement
 58            else
 59                FrictionCondition=1;%if the Friction condition is 1 there is movement
 60            end
 61        end
 62    %............................................................................
 63    %............................................................................
 64    %Calculates the acceleration components in X and Y for the centre of
 65    %gravity of the object and the angular acceleration
 66        function GeneralCoordinatesDoubleDot = Equation_Of_Motion(obj,GenCordExt)
 67            GeneralCoordinatesDoubleDot=zeros(3);
 68            %Ecuation of motion for X coordinates
 69            SignOposite=-1*sign(GenCordExt);
 70            if(NonLinearFriction(obj,abs(GenCordExt(1)),'F_s_x')== 1)
 71                GeneralCoordinatesDoubleDot(1)= (GenCordExt(1)+ SignOposite(1)*obj.F_k_x)/obj.M;
 72            else
 73                GeneralCoordinatesDoubleDot(1)= 0;
 74            end
 75        %Ecuation of motion for y coordinates
 76            if(NonLinearFriction(obj,abs(GenCordExt(2)),'F_s_y')== 1)
 77                GeneralCoordinatesDoubleDot(2)= (GenCordExt(2)+ SignOposite(2)*obj.F_k_y)/obj.M;
 78            else
 79                GeneralCoordinatesDoubleDot(2)= 0;
 80            end
 81        %Ecuation of motion for Tetha coordinates
 82            if(NonLinearFriction(obj,abs(GenCordExt(3)),'T_s')== 1)
 83                GeneralCoordinatesDoubleDot(3)= (GenCordExt(3)+ SignOposite(3)*obj.T_k)/obj.Iz;
 84            else
 85                GeneralCoordinatesDoubleDot(3)= 0;
 86            end
 87        end
 88    %............................................................................
 89    %............................................................................
 90    %Absolute positions for G3 and P3 updated from frame 0
 91        function Positions = PosCalculation(obj,angle,xPositionG3,yPositionG3)
 92            Rotation_o_3=[cos(angle),-sin(angle);sin(angle),cos(angle)];
 93            Position_G3_P3_frame_0=Rotation_o_3*obj.Position_G3_P3_frame_cylinder;
 94            Position_o_P3=[xPositionG3;yPositionG3] + Position_G3_P3_frame_0;
 95            Positions=[obj.Position_o_G3,Position_o_P3];
 96        end
 97    %............................................................................
 98    %............................................................................
 99    %Absolute velocities for G3 and P3 updated from frame 0
100        function Velocities = VelCalculation (obj,angle,angleDot,xDotG3,yDotG3)
101            Velocity_o_G3_frame_0=[xDotG3;yDotG3];
102            Velocity_o_P3_frame_0=[xDotG3-obj.R*cos(angle)*angleDot;yDotG3-obj.R*sin(angle)*angleDot];
103            Velocities=[Velocity_o_G3_frame_0,Velocity_o_P3_frame_0];
104        end
105    %............................................................................
106    %............................................................................
107    %Absolute Accelerations for G3 and P3 updated from frame 0

108        function Accelerations = AccCalculation (obj,angle,angleDot,angleDoubleDot,xDoubleDot,yDoubleDot)
109            Acceleration_o_G3_frame_0=[xDoubleDot;yDoubleDot];
110            Acceleration_o_P3_frame_0=[xDoubleDot+obj.R*sin(angle)*(angleDot^2)-obj.R*cos(angle)*angleDoubleDot;...
111                                       yDoubleDot-obj.R*cos(angle)*(angleDot^2)-obj.R*sin(angle)*angleDoubleDot];
112            Accelerations=[Acceleration_o_G3_frame_0,Acceleration_o_P3_frame_0,angleDoubleDot];
113        end

114    end
115    methods (Access = protected)
116    %............................................................................
117    %............................................................................
118    %Function that is called in Simulink to calculate updated
119    %positions,velocities and accelerations for the cylinder
120        function CylinderResults=stepImpl(obj,FextX,FextY,TorqueExt,xDotG3,yDotG3,angleDot,...
121                                                        xPositionG3,yPositionG3,angle)
122            %obj.Urk=Urk;
123            %obj.T_k=(2/3)*obj.Urk*obj.R*obj.M*obj.g;
124            GenCordExt=[FextX,FextY,TorqueExt];
125            GeneralCoordinatesDoubleDot = Equation_Of_Motion(obj,GenCordExt);
126            xDoubleDot = GeneralCoordinatesDoubleDot(1);
127            yDoubleDot = GeneralCoordinatesDoubleDot(2);
128            angleDoubleDot = GeneralCoordinatesDoubleDot(3);
129            Positions = PosCalculation(obj,angle,xPositionG3,yPositionG3);
130            Velocities = VelCalculation (obj,angle,angleDot,xDotG3,yDotG3);
131            Accelerations = AccCalculation (obj,angle,angleDot,angleDoubleDot,xDoubleDot,yDoubleDot);
132            CylinderResults = [Accelerations,Positions(3),Positions(4),obj.R];
133        end
134    end
135end
