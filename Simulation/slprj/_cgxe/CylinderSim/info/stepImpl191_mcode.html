 25    %Constructor
 26        function obj=Cylinder(varargin)
 27            obj.R=0.02;
 28            obj.M=0.003;
 29            obj.g=9.81;
 30            obj.Us=0.5;
 31            obj.Uk=0.48;
 32            obj.Urs=0.5;
 33            obj.Urk=0.48;
 34            obj.Ucs=0.5;
 35            obj.Uck=0.4;
 36            obj.Iz=(obj.M*obj.R^2)/2;
 37            obj.F_s_x=obj.M*obj.g*obj.Us;
 38            obj.F_s_y=obj.M*obj.g*obj.Us;
 39            obj.F_k_x=obj.M*obj.g*obj.Uk;
 40            obj.F_k_y=obj.M*obj.g*obj.Uk;
 41            obj.T_s=(2/3)*obj.Urs*obj.R*obj.M*obj.g;
 42            obj.T_k=(2/3)*obj.Urk*obj.R*obj.M*obj.g;
 43            obj.Position_o_G3=[0;0];
 44            obj.Position_G3_P3_frame_cylinder=[0;obj.R];
 45        end

 48    %Evaluates if the external force is greater than the static
 49    %friction forces or torques
 50        function FrictionCondition=NonLinearFriction(obj,GenCordExt,Election)
 51            %Evaluates wich static friction have to be compared
 52            if (strcmp(Election,'F_s_x')==1)
 53                ForceEvaluation=obj.F_s_x;
 54            elseif(strcmp(Election,'F_s_y')==1)
 55                ForceEvaluation=obj.F_s_y;
 56            else
 57                ForceEvaluation=obj.T_s;
 58            end
 59            %Condition checker
 60            if(GenCordExt<=ForceEvaluation)
 61                FrictionCondition=0;%if the Friction condition is 0 there is no movement
 62            else
 63                FrictionCondition=1;%if the Friction condition is 1 there is movement
 64            end
 65        end

 67    %............................................................................
 68    %Calculates the acceleration components in X and Y for the centre of
 69    %gravity of the object and the angular acceleration
 70        function GeneralCoordinatesDoubleDot = Equation_Of_Motion(obj,GenCordExt)
 71            GeneralCoordinatesDoubleDot=zeros(3);
 72            %Ecuation of motion for X coordinates
 73            SignOposite=-1*sign(GenCordExt);
 74            if(NonLinearFriction(obj,abs(GenCordExt(1)),'F_s_x')== 1)
 75                GeneralCoordinatesDoubleDot(1)= (GenCordExt(1)+ SignOposite(1)*obj.F_k_x)/obj.M;
 76            else
 77                GeneralCoordinatesDoubleDot(1)= 0;
 78            end
 79        %Ecuation of motion for y coordinates
 80            if(NonLinearFriction(obj,abs(GenCordExt(2)),'F_s_y')== 1)
 81                GeneralCoordinatesDoubleDot(2)= (GenCordExt(2)+ SignOposite(2)*obj.F_k_y)/obj.M;
 82            else
 83                GeneralCoordinatesDoubleDot(2)= 0;
 84            end
 85        %Ecuation of motion for Tetha coordinates
 86            if(NonLinearFriction(obj,abs(GenCordExt(3)),'T_s')== 1)
 87                GeneralCoordinatesDoubleDot(3)= (GenCordExt(3)+ SignOposite(3)*obj.T_k)/obj.Iz;
 88            else
 89                GeneralCoordinatesDoubleDot(3)= 0;
 90            end
 91        end

 94    %Absolute positions for G3 and P3 updated from frame 0
 95        function Positions = PosCalculation(obj,angle,xPositionG3,yPositionG3)
 96            Rotation_o_3=[cos(angle),-sin(angle);sin(angle),cos(angle)];
 97            Position_G3_P3_frame_0=Rotation_o_3*obj.Position_G3_P3_frame_cylinder;
 98            Position_o_P3=[xPositionG3;yPositionG3] + Position_G3_P3_frame_0;
 99            Positions=[obj.Position_o_G3,Position_o_P3];
100        end

103    %Absolute velocities for G3 and P3 updated from frame 0
104        function Velocities = VelCalculation (obj,angle,angleDot,xDotG3,yDotG3)
105            Velocity_o_G3_frame_0=[xDotG3;yDotG3];
106            Velocity_o_P3_frame_0=[xDotG3-obj.R*cos(angle)*angleDot;yDotG3-obj.R*sin(angle)*angleDot];
107            Velocities=[Velocity_o_G3_frame_0,Velocity_o_P3_frame_0];
108        end
109    %............................................................................
110    %............................................................................
111    %Absolute Accelerations for G3 and P3 updated from frame 0
112        function Accelerations = AccCalculation (obj,angle,angleDot,angleDoubleDot,xDoubleDot,yDoubleDot)
113            Acceleration_o_G3_frame_0=[xDoubleDot;yDoubleDot];
114            Acceleration_o_P3_frame_0=[xDoubleDot+obj.R*sin(angle)*(angleDot^2)-obj.R*cos(angle)*angleDoubleDot;...
115                                       yDoubleDot-obj.R*cos(angle)*(angleDot^2)-obj.R*sin(angle)*angleDoubleDot];
116            Accelerations=[Acceleration_o_G3_frame_0,Acceleration_o_P3_frame_0,[angleDoubleDot;0]];
117        end

120    %Absolute Accelerations for G3 and P3 updated from frame 0
121        function RotMatrix_0_4=RotationToContactFrame(obj,CollisionPosition,xPositionG3,yPositionG3)
122            if ((CollisionPosition(1)-xPositionG3)==0)
123                if((CollisionPosition(2)-yPositionG3)>0)
124                    Theta=pi/2;
125                else
126                    Theta=3*pi/2;
127                end
128            else
129                tan_angle=(CollisionPosition(2)-yPositionG3)/(CollisionPosition(1)-xPositionG3);
130                if(tan_angle>0 && (CollisionPosition(1)-xPositionG3)>0 && (CollisionPosition(2)-yPositionG3)>0)
131                    Theta=atan(tan_angle);
132                elseif(tan_angle>0 && (CollisionPosition(1)-xPositionG3)<0 && (CollisionPosition(2)-yPositionG3)<0)
133                    Theta=atan(tan_angle)+pi;
134                elseif(tan_angle<0 && (CollisionPosition(1)-xPositionG3)<0 && (CollisionPosition(2)-yPositionG3)>0)
135                    Theta=atan(tan_angle);
136                elseif(tan_angle<0 && (CollisionPosition(1)-xPositionG3)>0 && (CollisionPosition(2)-yPositionG3)<0)
137                    Theta=atan(tan_angle)+pi;
138                elseif(tan_angle==0)
139                    Theta=0;
140                end
141            end 
142            RotMatrix_0_4=[cos(Theta) -sin(Theta);sin(Theta) cos(Theta)];
143        end

188            if(CollisionCondition(1)==1 && CollisionCondition(2)==1)
189                BunchOfTotalForce=[IndexForce,ThumbForce];
190            elseif(CollisionCondition(1)==0 && CollisionCondition(2)==1)
191                BunchOfTotalForce=[[0;0],ThumbForce];
192            elseif(CollisionCondition(1)==1 && CollisionCondition(2)==0)
193                BunchOfTotalForce=[IndexForce,[0;0]];
194            else
195                BunchOfTotalForce=[[0;0],[0;0]];
196            end

204        function CylinderResults=stepImpl(obj,Urk,IndexFext,ThumbFext,CollisionCondition,CollisionPosition,...
205                                            xDotG3,yDotG3,angleDot,xPositionG3,yPositionG3,angle)
206            obj.Urk=Urk;
207            obj.T_k=(2/3)*obj.Urk*obj.R*obj.M*obj.g;
208            
209            GenCordExt=[FextX,FextY,TorqueExt];
210            FrictionCondition=NonLinearFriction(obj,abs(GenCordExt(3)),'T_s');
211            GeneralCoordinatesDoubleDot = Equation_Of_Motion(obj,GenCordExt);
212            xDoubleDot = GeneralCoordinatesDoubleDot(1);
213            yDoubleDot = GeneralCoordinatesDoubleDot(2);
214            angleDoubleDot = GeneralCoordinatesDoubleDot(3);
215            Positions = PosCalculation(obj,angle,xPositionG3,yPositionG3);
216            Velocities = VelCalculation (obj,angle,angleDot,xDotG3,yDotG3);
217            Accelerations = AccCalculation (obj,angle,angleDot,angleDoubleDot,xDoubleDot,yDoubleDot);
218            CylinderResults = [Accelerations([1 2 5]),Positions([3 4]),obj.R,obj.Urk,FrictionCondition];
219        end
