1classdef Phalanx < handle
 2    % Phalanx Class
 3    % Represents the Phalanx of an underactuated Finger.
 4    properties (SetAccess = immutable)
 5        % Physical Phalanx properties
 6        k = 1;      % Joint stiffness (Nm/rad)
 7        l = 0.1;    % Link length (m)
 8        r = 0.02;   % Pulley radius (m)
 9        m = 0.02;   % Link mass (kg)
10        I           % Link z-inertia
11    end
12    properties
13        % Physical finger properties
14        miuC = 0.8; % Coefficient of friction between phalanx and contact object.
15        miuE = 0.8; % Coefficient of friction between phalanx and disturbance object.% Kinematic properties
16        theta0 = pi/10; % Joint rest angle (rad)
17        theta = pi/10; % Joint angle (rad)
18        thetaDot = 0; % First derivative (velocity) of joint angle (rad/s)
19        thetaDotDot = 0; % Second derivative (acceleration) of joint angle (rad/s^2)
20        % Contact Kinematics properties
21        a = 0; % Distance from joint to contact force (m)
22        fc % Contact Force
23        % Disturbance Kinematics properties
24        b = 0; % Distance from joint to disturbance force (m)
25        fe % Disturbance Force
26    end
27    methods

28        function obj = Phalanx(k, l, r, m, theta, linkNum)
29            % Constructor with parameter initialization
30            
31            % Custom initialization
32            if(nargin ~= 0)
33                obj.k = k;
34                obj.l = l;
35                obj.r = r;
36                obj.m = m;
37                obj.theta = theta;
38                if (linkNum>1)
39                    obj.I = obj.m*(obj.l^2)/12;
40                else
41                    obj.I = obj.m*(obj.l^2)/3;
42                end
43            else
44                obj.I = obj.m*(obj.l^2)/3;
45            end
46            obj.theta0 = obj.theta;
47            obj.fc = ForceNormal(0,obj.a,obj.miuC);
48            obj.fe = ForceNormal(0,obj.b,obj.miuE);
49        end

50    end
51end
